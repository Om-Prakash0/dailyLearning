2028. Find Missing Observations








class Solution {
public:
    vector<int> missingRolls(vector<int>& rolls, int mean, int n) {
        int m = rolls.size();
        int sum = 0;

        for (auto a : rolls) {
            sum += a;
        }

        int total = mean * (n + m);
        int missingSum = total - sum;

        if (missingSum < n || missingSum > 6 * n) {
            return {};
        }

        vector<int> missing(n, 1); 
        missingSum -= n;           
        int i = 0;
        while (missingSum > 0) {
            int add = min(5, missingSum); 
            missing[i] += add;
            missingSum -= add;
            i++;
        }

        return missing;
    }
};
