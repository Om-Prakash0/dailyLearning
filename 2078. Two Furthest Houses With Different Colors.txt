2078. Two Furthest Houses With Different Colors


class Solution {
public:
    int maxDistance(vector<int>& colors) {
       
        int ele=colors.size();
        int ans=INT_MIN;
        for(int i=0;i<ele;i++){
            for(int j=ele-1;j>=0;j--){
                if(colors[i]!=colors[j]){
                int dif=abs(j-i);
                ans=max(dif,ans);
                break;
                }
            }
        }
        return ans;
    }
};
