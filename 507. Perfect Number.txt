507. Perfect Number



class Solution {
public:
    bool checkPerfectNumber(int num) {
        vector<int>arr;
        for(int i=1;i<num;i++){
            if(num%i==0){
                arr.push_back(i);
            }
        }
        int sum=0;
        for(auto a:arr){
            sum+=a;
        }
        return sum==num;
    }
};
