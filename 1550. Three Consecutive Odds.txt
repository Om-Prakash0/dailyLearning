1550. Three Consecutive Odds





class Solution {
public:
    bool threeConsecutiveOdds(vector<int>& arr) {
        int n=arr.size();
        bool flag=false;
        for(int i=0;i<=n-3;i++){
            int a=arr[i];
            int b=arr[i+1];
            int c=arr[i+2];
            if(a%2!=0 && b%2!=0 && c%2!=0){
                flag=true;
                break;
            }
        }
        return flag;    }
};
