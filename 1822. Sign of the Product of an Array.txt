1822. Sign of the Product of an Array

class Solution {
public:
    int arraySign(vector<int>& nums) {
        int n=nums.size();
        int product=1;
        for(int i=0;i<n;i++){
           if(nums[i]<0){
                product*=-1;
            }
            else if(nums[i]>0){
                product*=1;
            }
            else{
                product*=0;
            }
        }
        return product;
    }
};
